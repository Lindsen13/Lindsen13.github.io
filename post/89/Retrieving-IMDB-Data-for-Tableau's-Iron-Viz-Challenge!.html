Retrieving IMDB Data for Tableau's Iron Viz Challenge!

2020-08-03 • By Ivo

Scraping and Cleaning data, the quick-and-dirty method.

Most of the time, I use Tableau to visualize my data. I recently found out that the submission deadline for the 2020 content was comming up. I decided to join as well! For this, I needed a dataset.

This post will just show my quick-and-dirty Python script to scrape all the movies from IMDB. I will not go over the script to descibe what I have done; since the idea of this post is just to showcase how skills in Python can be used to quicky mock up a script that will help you with a one-time-only task.

import requests
from bs4 import BeautifulSoup
import re
import pandas as pd

# Paramters to initialize the script:
page_start = 1
genre = 'action'
url = f'https://www.imdb.com/search/title/?title_type=feature&num_votes=25000,&genres={genre}&view=simple&sort=alpha,asc&start={page_start}&ref_=adv_nxt'

output = requests.get(url)

soup = BeautifulSoup(output.text, 'html.parser')
soup.find(class_='desc')
results = re.search('d+,d+', soup.find(class_='desc').text)
number_of_results = int(results.group(0).replace(',',''))

# Function to clean the image url
def clean_image_url(image_div):
    try:
        image_url = re.search('loadlate=".*" src=',str(image_div.select('a')[0])).group(0)
        image_url = image_url.replace('loadlate=','')
        image_url = image_url.replace('src=','')
        image_url = image_url.strip()
        image_url = image_url.replace('"','')
    except:
        image_url = None
        print('issue with image')
    return image_url

# Function to clean the votes
def clean_votes(votes_div):
    try:
        votes = votes_div.select('strong')[0]['title']
        votes = re.search('base on .* votes',votes)
        votes = votes.group(0)
        votes = votes.replace('base on ','')
        votes = votes.replace(' votes','')
        votes = votes.replace(',','')
        votes = int(votes)
    except:
        votes = None
        print('issue with votes')
    return votes

# Function to clean the year
def clean_year(year):
    try:
        year = year.text
        year = year.replace('(I) ','').replace('(II) ','').replace('(III) ','')
        year = year.replace(')','')
        year = year.replace('(','')
        year = int(year)
    except:
        year = None
        print('issue with year')
    return year

# Function to clean the title
def clean_title(title): 
    try:
        title = title.select('a')[0].text
    except:
        title = None
        print('issue with title')
    return title

# Function to clean the actor: from string to list, split by a comma.
def clean_actor(actor):
    try:
        actor = actor.find_all('span')[1]['title'].split(',')
    except:
        actor = None
        print('issue with actor')
    return actor

# Function to clean the rank
def clean_rank(rank):
    try:
        rank = rank.text.replace('.','')
    except:
        rank = None
        print('issue with rank')
    return rank

# Function to clean the rating
def clean_rating(rating):
    try:
        rating = rating.text.strip()
        rating = float(rating)
    except:
        rating = None
        print('issue with rating')
    return rating

# Loop over all genres
film_list = []
for genre in ['Action', 'Adventure', 'Animation', 'Biography', 'Comedy', 'Crime', 'Drama', 'Family', 'Fantasy', 'Film-Noir', 'History', 'Horror', 'Music', 'Musical', 'Mystery', 'Romance', 'Sci-Fi', 'Sport', 'Thriller', 'War', 'Western']:
    # Retrieve number of films within genre, to determine number of loops (50 films a page)
    page_start = 1
    url = f'https://www.imdb.com/search/title/?title_type=feature&num_votes=25000,&genres={genre}&view=simple&sort=alpha,asc&start={page_start}&ref_=adv_nxt'
    output = requests.get(url)
    soup = BeautifulSoup(output.text, 'html.parser')
    soup.find(class_='desc')
    """
    Everything is wrapped in a try-except block. Doing this, we miss some information.
    But for my puspose, that does not matter.
    """
    try:
        try:
            results = re.search('d+,d+', soup.find(class_='desc').text)
            number_of_results = int(results.group(0).replace(',',''))
        except:
            results = re.search('d{3}', soup.find(class_='desc').text)
            number_of_results = int(results)
        print(f'{str(int(number_of_results/50))} pages to be scraped for {genre}')
        temp_i = 0
        while page_start < number_of_results:
            url = f'https://www.imdb.com/search/title/?title_type=feature&num_votes=25000,&genres={genre}&view=simple&sort=alpha,asc&start={page_start}&ref_=adv_nxt'
            output = requests.get(url)
            soup = BeautifulSoup(output.text, 'html.parser')
            soup_list = soup.find_all(class_='lister-item mode-simple')
            for film in soup_list:
                #Append each film to the list
                film_list.append({
                    'title': clean_title(film.find("span", class_='lister-item-header')),
                    'actors': clean_actor(film.find("span", class_='lister-item-header')),
                    'image': clean_image_url(film.find(class_='lister-item-image')),
                    'rank': clean_rank(film.find("span", class_='lister-item-index unbold text-primary')),
                    'year': clean_year(film.find("span", class_='lister-item-year text-muted unbold')),
                    'votes': clean_votes(film.find(class_='col-imdb-rating')),
                    'rating': clean_rating(film.find(class_='col-imdb-rating')),
                    'genre': genre
                })
            # Logic for checking process
            temp_i += 1
            if temp_i %250 == 0:
                print(f'scraped {str(len(film_list))} movies')
            page_start += 50
    except:
        # If something goes wrong, maximum 50 films are excluded and the next batch is processed
        continue
output = eval(film_list)

# Data cleaning, generating columns for each actor/director
for i in range(len(output)):
    if output[i]['year'] != None:
        output[i]['year'] = int(output[i]['year'])
    if len(output[i]['actors']) == 0 :
        continue
    elif len(output[i]['actors']) == 1 :
        output[i]['1st_name'] = output[i]['actors'][0].replace(' (dir.)','')
    elif len(output[i]['actors']) == 2 :    
        output[i]['1st_name'] = output[i]['actors'][0].replace(' (dir.)','')
        output[i]['2nd_name'] = output[i]['actors'][1].replace(' (dir.)','')
    else:
        output[i]['1st_name'] = output[i]['actors'][0].replace(' (dir.)','')
        output[i]['2nd_name'] = output[i]['actors'][1].replace(' (dir.)','')
        output[i]['3rd_name'] = output[i]['actors'][2].replace(' (dir.)','')

# To pandas dataframe, and some more cleaning.
df = pd.DataFrame(output)
df = df.drop('actors', 1)
df = df.drop_duplicates(['title','rank','year'],keep='first')

# Output CSV, ready to be read with Tableau!
df.to_csv('output.csv',index=False)
